    # Maven

    <repository>
        <name>Enaium</name>
        <id>Enaium</id>
        <url>https://enaium.gitee.io/maven/</url>
    </repository>
</repositories>

<dependencies>
    <dependency>
        <groupId>cn.enaium</groupId>

    </dependency>
</dependencies>
